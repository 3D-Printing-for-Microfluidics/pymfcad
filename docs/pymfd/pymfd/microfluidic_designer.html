 33class _InstantiationTrackerMixin:
 34    """Mixin class to track where a component was instantiated."""
 35
 36    def __init__(self):
 37        # Get the first relevant frame outside of this class
 38        for frame_info in inspect.stack():
 39            filename = frame_info.filename
 40            if "site-packages" in filename or filename == __file__:
 41                continue
 42            self._instantiation_path = Path(filename).resolve()
 43            break
 44
 45    @property
 46    def instantiation_dir(self) -> Path:
 47        """Return (directory, filename_stem) of the file that instantiated the component, if it's a Device or Component."""
 48        class_name = type(self).__name__
 49
 50        if class_name in {"Device", "Component"}:
 51            return self._instantiation_path.parent
 52
 53        # Fallback: use where the class is defined
 54        module_name = self.__class__.__module__
 55        module = sys.modules.get(module_name) or importlib.import_module(module_name)
 56        path = Path(module.__file__).resolve()
 57        return path.parent
 58
 59    @property
 60    def instantiating_file_stem(self) -> str:
 61        """Return (directory, filename_stem) of the file that instantiated the component, if it's a Device or Component."""
 62        class_name = type(self).__name__
 63
 64        if class_name in {"Device", "Component"}:
 65            return self._instantiation_path.stem
 66
 67        # Fallback: use where the class is defined
 68        module_name = self.__class__.__module__
 69        module = sys.modules.get(module_name) or importlib.import_module(module_name)
 70        path = Path(module.__file__).resolve()
 71        return path.stem

 74class Port(_InstantiationTrackerMixin):
 75    """
 76    Class representing a port in a microfluidic device.
 77    Ports are used to connect components and define their interaction with the environment.
 78    Each port has a type (IN, OUT, INOUT), a position in 3D space, a size, and a surface normal.
 79    The surface normal defines the direction in which the port is oriented.
 80    """
 81
 82    class PortType(Enum):
 83        """Enumeration for port types."""
 84
 85        IN = 1
 86        OUT = 2
 87        INOUT = 3
 88
 89    class SurfaceNormal(Enum):
 90        """Enumeration for surface normals."""
 91
 92        POS_X = 1
 93        POS_Y = 2
 94        POS_Z = 3
 95        NEG_X = 4
 96        NEG_Y = 5
 97        NEG_Z = 6
 98
 99    _vector_map = {
100        SurfaceNormal.POS_X: (1, 0, 0),
101        SurfaceNormal.POS_Y: (0, 1, 0),
102        SurfaceNormal.POS_Z: (0, 0, 1),
103        SurfaceNormal.NEG_X: (-1, 0, 0),
104        SurfaceNormal.NEG_Y: (0, -1, 0),
105        SurfaceNormal.NEG_Z: (0, 0, -1),
106    }
107
108    def __init__(
109        self,
110        _type: PortType,
111        position: tuple[int, int, int],
112        size: tuple[int, int, int],
113        surface_normal: SurfaceNormal,
114    ):
115        """
116        Initialize a port.
117
118        Parameters:
119        - _type (PortType): The type of the port (IN, OUT, INOUT).
120        - position (tuple[int, int, int]): The position of the port in 3D space.
121        - size (tuple[int, int, int]): The size of the port.
122        - surface_normal (SurfaceNormal): The surface normal of the port, defining its orientation.
123        """
124        super().__init__()
125        self._parent = None
126        self._name = None
127        self._type = _type
128        self._position = position
129        self._size = size
130        self._surface_normal = surface_normal
131
132    def get_name(self) -> str:
133        """Get the name of the port."""
134        if self._name is None:
135            raise ValueError(f"Port has not been named")
136        else:
137            return f"{self._parent._name}_{self._name}"
138
139    def get_fully_qualified_name(self) -> str:
140        """Get the fully qualified name of the port, including parent component names."""
141        if self._name is None:
142            raise ValueError(f"Port has not been named")
143        name = self._name
144        parent = self._parent
145        while parent is not None:
146            if parent._name is not None:
147                name = f"{parent._name}.{name}"
148                parent = parent._parent
149            else:
150                name = f"{parent.instantiating_file_stem}.{name}"
151                return name
152        return name
153
154    def to_vector(self) -> tuple[int, int, int]:
155        """Convert the surface normal to a vector."""
156        try:
157            return self._vector_map[self._surface_normal]
158        except KeyError:
159            raise ValueError(f"Unsupported surface normal: {self._surface_normal}")
160
161    def get_bounding_box(self) -> tuple[int, int, int, int, int, int]:
162        """
163        Get the bounding box of the port.
164        The bounding box is defined by the position and size of the port,
165        adjusted based on the surface normal direction.
166
167        Returns:
168        - A tuple of six integers representing the bounding box coordinates:
169        (min_x, min_y, min_z, max_x, max_y, max_z)
170        """
171        dx, dy, dz = self._vector_map[self._surface_normal]
172        pos = list(self._position)
173        size = self._size
174
175        # For negative directions, shift start back by size
176        if dx < 0:
177            pos[0] -= size[0]
178        if dy < 0:
179            pos[1] -= size[1]
180        if dz < 0:
181            pos[2] -= size[2]
182
183        return (
184            pos[0],
185            pos[1],
186            pos[2],
187            pos[0] + size[0],
188            pos[1] + size[1],
189            pos[2] + size[2],
190        )
191
192    def get_origin(self):
193        """Get the origin of the port, which is the minimum corner of its bounding box."""
194        return self.get_bounding_box()[0:3]
195
196    def get_color(self):
197        """
198        Get the color of the port based on its type.
199
200        The color is determined as follows:
201        - IN ports are green
202        - OUT ports are red
203        - INOUT ports are blue
204        - If the type is not recognized, it defaults to white.
205
206        Returns:
207        - Color: The color of the port.
208        """
209        if self._type == Port.PortType.IN:
210            return Color.from_name("green", 255)  # Green
211        elif self._type == Port.PortType.OUT:
212            return Color.from_name("red", 255)  # Red
213        elif self._type == Port.PortType.INOUT:
214            return Color.from_name("blue", 255)  # Blue
215        else:
216            return Color.from_name("white", 255)  # White

506    def rotate(self, rotation: int, in_place: bool = False) -> Component:
507        if rotation % 90 != 0:
508            raise ValueError("Rotation must be a multiple of 90 degrees")
509
510        if in_place:
511            original_position = self._position
512            # Translate the component to position for in-place rotation
513            self.translate(
514                (-self._position[0], -self._position[1], -self._position[2]),
515                set_origin=False,
516            )
517
518        for component in self.subcomponents:
519            component.rotate(rotation)
520
521        for shape in self.shapes:
522            shape.rotate((0, 0, rotation))
523
524        rot = rotation % 360
525
526        # Mapping for 90-degree rotation steps around Z axis
527        vector_rotation_map = {
528            90: {
529                Port.SurfaceNormal.POS_X: (Port.SurfaceNormal.POS_Y, (-1, 0)),
530                Port.SurfaceNormal.POS_Y: (Port.SurfaceNormal.NEG_X, (0, 0)),
531                Port.SurfaceNormal.NEG_X: (Port.SurfaceNormal.NEG_Y, (-1, 0)),
532                Port.SurfaceNormal.NEG_Y: (Port.SurfaceNormal.POS_X, (0, 0)),
533            },
534            180: {
535                Port.SurfaceNormal.POS_X: (Port.SurfaceNormal.NEG_X, (0, -1)),
536                Port.SurfaceNormal.POS_Y: (Port.SurfaceNormal.NEG_Y, (-1, 0)),
537                Port.SurfaceNormal.NEG_X: (Port.SurfaceNormal.POS_X, (0, -1)),
538                Port.SurfaceNormal.NEG_Y: (Port.SurfaceNormal.POS_Y, (-1, 0)),
539            },
540            270: {
541                Port.SurfaceNormal.POS_X: (Port.SurfaceNormal.NEG_Y, (0, 0)),
542                Port.SurfaceNormal.POS_Y: (Port.SurfaceNormal.POS_X, (0, -1)),
543                Port.SurfaceNormal.NEG_X: (Port.SurfaceNormal.POS_Y, (0, 0)),
544                Port.SurfaceNormal.NEG_Y: (Port.SurfaceNormal.NEG_X, (0, -1)),
545            },
546        }
547
548        for port in self.ports:
549            x, y, z = port._position
550            size_x, size_y = port._size[0], port._size[1]
551
552            if rot == 90:
553                port._position = (-y, x, z)
554            elif rot == 180:
555                port._position = (-x, -y, z)
556            elif rot == 270:
557                port._position = (y, -x, z)
558
559            if port._surface_normal in vector_rotation_map.get(rot, {}):
560                new_vector, (dx, dy) = vector_rotation_map[rot][port._surface_normal]
561                port._position = (
562                    port._position[0] + dx * port._size[0],
563                    port._position[1] + dy * port._size[1],
564                    port._position[2],
565                )
566                port._surface_normal = new_vector
567
568            # Adjust Z ports if needed
569            if port._surface_normal in (
570                Port.SurfaceNormal.POS_Z,
571                Port.SurfaceNormal.NEG_Z,
572            ):
573                if rot == 90:
574                    port._position = (
575                        port._position[0] - port._size[0],
576                        port._position[1],
577                        port._position[2],
578                    )
579                elif rot == 180:
580                    port._position = (
581                        port._position[0] - port._size[0],
582                        port._position[1] - port._size[1],
583                        port._position[2],
584                    )
585                elif rot == 270:
586                    port._position = (
587                        port._position[0],
588                        port._position[1] - port._size[1],
589                        port._position[2],
590                    )
591
592        if in_place:
593            # Translate the component so new negative-negative corner is at original position
594            if rot == 90:
595                self.translate(
596                    (
597                        original_position[0] + self._size[1],
598                        original_position[1],
599                        original_position[2],
600                    ),
601                    set_origin=False,
602                )
603            elif rot == 180:
604                self.translate(
605                    (
606                        original_position[0] + self._size[0],
607                        original_position[1] + self._size[1],
608                        original_position[2],
609                    ),
610                    set_origin=False,
611                )
612            elif rot == 270:
613                self.translate(
614                    (
615                        original_position[0],
616                        original_position[1] + self._size[0],
617                        original_position[2],
618                    ),
619                    set_origin=False,
620                )
621
622        return self

624    def mirror(
625        self, mirror_x: bool = False, mirror_y: bool = False, in_place: bool = False
626    ) -> Component:
627        if not mirror_x and not mirror_y:
628            return self  # No mirroring requested
629
630        if mirror_x and mirror_y:
631            return self.rotate(180, in_place=in_place)
632
633        if in_place:
634            original_position = self._position
635            # Translate the component to position for in-place mirroring
636            self.translate(
637                (-self._position[0], -self._position[1], -self._position[2]),
638                set_origin=False,
639            )
640
641        for component in self.subcomponents:
642            component.mirror(mirror_x, mirror_y)
643
644        for shape in self.shapes:
645            shape.mirror((mirror_x, mirror_y, False))
646
647        # Surface normal flips
648        mirror_vector_map = {
649            "x": {
650                Port.SurfaceNormal.POS_X: Port.SurfaceNormal.NEG_X,
651                Port.SurfaceNormal.NEG_X: Port.SurfaceNormal.POS_X,
652            },
653            "y": {
654                Port.SurfaceNormal.POS_Y: Port.SurfaceNormal.NEG_Y,
655                Port.SurfaceNormal.NEG_Y: Port.SurfaceNormal.POS_Y,
656            },
657        }
658
659        for port in self.ports:
660            x, y, z = port._position
661            sx, sy = port._size[0], port._size[1]
662
663            if mirror_x:
664                x = -x - sx
665                # If pointing in +X or -X, correct for sticking out
666                if port._surface_normal == Port.SurfaceNormal.POS_X:
667                    x += sx
668                elif port._surface_normal == Port.SurfaceNormal.NEG_X:
669                    x += sx
670                port._surface_normal = mirror_vector_map["x"].get(
671                    port._surface_normal, port._surface_normal
672                )
673
674            if mirror_y:
675                y = -y - sy
676                # If pointing in +Y or -Y, correct for sticking out
677                if port._surface_normal == Port.SurfaceNormal.POS_Y:
678                    y += sy
679                elif port._surface_normal == Port.SurfaceNormal.NEG_Y:
