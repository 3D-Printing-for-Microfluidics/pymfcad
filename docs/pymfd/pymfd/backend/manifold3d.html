 45    def _translate_keepouts(self, translation: tuple[float, float, float]) -> None:
 46        """
 47        ###### Translate the keepouts.
 48
 49        ###### Parameters:
 50        - translation (tuple[float, float, float]): The translation.
 51        """
 52        dx, dy, dz = translation
 53        self._keepouts = [
 54            [x0 + dx, y0 + dy, z0 + dz, x1 + dx, y1 + dy, z1 + dz]
 55            for x0, y0, z0, x1, y1, z1 in self._keepouts
 56        ]

 58    def _rotate_point(
 59        self, point: tuple[float, float, float], rotation: tuple[float, float, float]
 60    ) -> None:
 61        """
 62        ###### Rotate a point around origin (0,0,0) with Euler angles (in degrees) in XYZ order.
 63
 64        ###### Parameters:
 65        - point (tuple[float, float, float]): The point to rotate.
 66        - rotation (tuple[float, float, float]): The rotation.
 67        """
 68        # Convert degrees to radians
 69        rx, ry, rz = np.radians(rotation)
 70        x, y, z = point
 71
 72        # Rotate around X
 73        cos_rx, sin_rx = np.cos(rx), np.sin(rx)
 74        y, z = y * cos_rx - z * sin_rx, y * sin_rx + z * cos_rx
 75
 76        # Rotate around Y
 77        cos_ry, sin_ry = np.cos(ry), np.sin(ry)
 78        x, z = x * cos_ry + z * sin_ry, -x * sin_ry + z * cos_ry
 79
 80        # Rotate around Z
 81        cos_rz, sin_rz = np.cos(rz), np.sin(rz)
 82        x, y = x * cos_rz - y * sin_rz, x * sin_rz + y * cos_rz
 83
 84        return [x, y, z]

 86    def _rotate_keepouts(self, rotation: tuple[float, float, float]):
 87        """
 88        ###### Rotate the keepouts.
 89
 90        ###### Parameters:
 91        - rotation (tuple[float, float, float]): The rotation.
 92        """
 93        rotated_keepouts = []
 94        for x0, y0, z0, x1, y1, z1 in self._keepouts:
 95            # Get all 8 corners
 96            corners = [[x, y, z] for x in (x0, x1) for y in (y0, y1) for z in (z0, z1)]
 97            rotated_corners = [self._rotate_point(pt, rotation) for pt in corners]
 98            xs, ys, zs = zip(*rotated_corners)
 99            rotated_keepouts.append(
100                [min(xs), min(ys), min(zs), max(xs), max(ys), max(zs)]
101            )
102        self._keepouts = rotated_keepouts

124    def _mirror_keepouts(self, axis: tuple[bool, bool, bool]) -> None:
125        """
126        ###### Mirror the keepouts.
127
128        ###### Parameters:
129        - axis (tuple[bool, bool, bool]): The axis to mirror.
130        """
131        flip_x, flip_y, flip_z = axis
132        new_keepouts = []
133        for x0, y0, z0, x1, y1, z1 in self._keepouts:
134            nx0, nx1 = sorted([-x0, -x1]) if flip_x else (x0, x1)
135            ny0, ny1 = sorted([-y0, -y1]) if flip_y else (y0, y1)
136            nz0, nz1 = sorted([-z0, -z1]) if flip_z else (z0, z1)
137            new_keepouts.append([nx0, ny0, nz0, nx1, ny1, nz1])
138        self._keepouts = new_keepouts

174    def resize(self, size: tuple[int, int, int]) -> "Shape":
175        """
176        ###### Resize the shape to a given size in px/layer space.
177
178        ###### Parameters:
179        - size (tuple[int, int, int]): The new size in px/layer space.
180
181        ###### Returns:
182        - self (Shape): The resized shape.
183        """
184        # if size if 0 set it near 0
185        if size[0] == 0:
186            size = (0.0001, size[1], size[2])
187        if size[1] == 0:
188            size = (size[0], 0.0001, size[2])
189        if size[2] == 0:
190            size = (size[0], size[1], 0.0001)
191        bounds = self._object.bounding_box()
192        # convert bounds to px/layer
193        bounds = [
194            bounds[0],
195            bounds[1],
196            bounds[2],
197            bounds[3],
198            bounds[4],
199            bounds[5],
200        ]
201        # calculate scale factors in px/layer space
202        sx = size[0] / (bounds[3] - bounds[0])
203        sy = size[1] / (bounds[4] - bounds[1])
204        sz = size[2] / (bounds[5] - bounds[2])
205
206        self._scale_keepouts((sx, sy, sz))
207        self._object = self._object.scale((sx, sy, sz))

251    def _intersect_boxes(self, box1, box2):
252        # Compute the intersection of two axis-aligned bounding boxes
253        x_min = max(box1[0], box2[0])
254        y_min = max(box1[1], box2[1])
255        z_min = max(box1[2], box2[2])
256        x_max = min(box1[3], box2[3])
257        y_max = min(box1[4], box2[4])
258        z_max = min(box1[5], box2[5])
259
260        # Check for non-empty intersection
261        if x_min < x_max and y_min < y_max and z_min < z_max:
262            return [x_min, y_min, z_min, x_max, y_max, z_max]
263        else:
264            return None

267        intersections = []

271                if inter:
272                    intersections.append(inter)

292    def hull(self, other: "Shape") -> "Shape":
293        """
294        ###### Create a convex hull of this shape and another shape.
295        ###### This method combines the keepouts of both shapes and creates a bridge between their bounding boxes.
296
297        ###### Parameters:
298        - other (Shape): The other shape to combine with.
299
300        ###### Returns:
301        - self (Shape): The resulting shape after creating the hull.
302        """
303        # Combine keepouts
304        self._keepouts.extend(other._keepouts)
305
306        # Get both bounding boxes
307        b1 = self._object.bounding_box()
308        b1 = (
309            b1[0],
310            b1[1],
311            b1[2],
312            b1[3],
313            b1[4],
314            b1[5],
315        )
316        b2 = other._object.bounding_box()
317        b2 = (
318            b2[0],
319            b2[1],
320            b2[2],
321            b2[3],
322            b2[4],
323            b2[5],
324        )
325
326        center1 = [(b1[0] + b1[3]) / 2, (b1[1] + b1[4]) / 2, (b1[2] + b1[5]) / 2]
327        center2 = [(b2[0] + b2[3]) / 2, (b2[1] + b2[4]) / 2, (b2[2] + b2[5]) / 2]
328        deltas = [abs(center1[i] - center2[i]) for i in range(3)]
329
330        # The separation axis is the one with the largest center delta
331        sep_axis = deltas.index(max(deltas))
332
333        # Create a bridge box between bounding boxes along the separation axis
334        bridge_min = [0, 0, 0]
335        bridge_max = [0, 0, 0]
336
337        for i in range(3):
338            if i == sep_axis:
339                bridge_min[i] = min(b1[i], b2[i])
340                bridge_max[i] = max(b1[i + 3], b2[i + 3])
341            else:
342                bridge_min[i] = min(b1[i], b2[i])
343                bridge_max[i] = max(b1[i + 3], b2[i + 3])
344
345        self._keepouts.append(
346            [
347                bridge_min[0],
348                bridge_min[1],
349                bridge_min[2],
350                bridge_max[0],
351                bridge_max[1],
352                bridge_max[2],
353            ]
354        )
355
356        self._object = Manifold.batch_hull([self._object, other._object])
357        return self

446class Cylinder(Shape):
447    """
448    ###### Manifold3D cylinder.
449    """
450
451    def __init__(
452        self,
453        height: int,
454        radius: float = None,
455        bottom_r: float = None,
456        top_r: float = None,
457        center_xy: bool = True,
458        center_z: bool = False,
459        fn: int = 0,
460    ):
461        """
462        ###### Parameters:
463        - height (int): Height of the cylinder in layer space.
464        - radius (float): Radius of the cylinder in px space.
465        - bottom_r (float): Bottom radius of the cylinder in px space.
466        - top_r (float): Top radius of the cylinder in px space.
467        - center_xy (bool): Whether to center the cylinder in XY plane.
468        - center_z (bool): Whether to center the cylinder in Z plane.
469        - fn (int): Number of facets for the circular segments.
470        """
471        super().__init__()
472
473        # only allow radiuses to be multiples of 0.5
474        if radius is not None:
475            if not _is_integer(radius * 2):
476                raise ValueError("Cylinder radius must be a multiple of 0.5")
477        if bottom_r is not None:
478            if not _is_integer(bottom_r * 2):
479                raise ValueError("Cylinder radius (bottom) must be a multiple of 0.5")
480        if top_r is not None:
481            if not _is_integer(top_r * 2):
482                raise ValueError("Cylinder radius (top) must be a multiple of 0.5")
483
484        # validate shape is fully constrained
485        bottom = bottom_r if bottom_r is not None else radius
486        top = top_r if top_r is not None else radius
487        if bottom is None or top is None:
488            raise ValueError("Either radius or bottom_r and top_r must be provided.")
489
490        if top % 2 != bottom % 2:
491            raise ValueError(
492                "Cylinder top and bottom radius must both be either even or odd."
493            )
494
495        bottom = bottom_r if bottom_r is not None else radius
496        top = top_r if top_r is not None else radius
497
498        xy = 0
499        z = 0
500        if center_z and height % 2 != 0:
501            print(
502                f" Centered cylinder z dimension is odd. Shifting 0.5 px to align with px grid"
503            )
504            z = 0.5
505        if height == 0:
506            height = 0.0001
507        if center_xy:
508            if top * 2 % 2 != 0:  # can check either to or bottom
509                print(
510                    f" Centered cylinder radius is odd. Shifting 0.5 px to align with px grid"
511                )
512                xy = 0.5
513            self._object = Manifold.cylinder(
514                height=height,
515                radius_low=bottom,
516                radius_high=top,
517                circular_segments=fn,
518                center=center_z,
519            ).translate((xy, xy, z))
520        else:
521            radius = max(bottom, top)
522            self._object = Manifold.cylinder(
523                height=height,
524                radius_low=bottom,
525                radius_high=top,
526                circular_segments=fn,
527                center=center_z,
528            ).translate((radius, radius, z))
529        self._add_bbox_to_keepout(self._object.bounding_box())

613        ###### Parameters:
614        - size (tuple[int, int, int]): Size of the rounded cube in px/layer space.

616        - center (bool): Whether to center the rounded cube at the origin.
617        - fn (int): Number of facets for the circular segments.
618        - _no_validation (bool): If True, skip validation checks for odd dimensions (internal use).
619        """

621
622        # shift half a pixel if odd and centered
623        x = 0
624        y = 0
625        z = 0
626        if center and not _no_validation:
627            if size[0] % 2 != 0:
628                print(
629                    f" Centered rounded cube x dimension is odd. Shifting 0.5 px to align with px grid"
630                )
631                x = 0.5
632            if size[1] % 2 != 0:
633                print(
634                    f" Centered rounded cube y dimension is odd. Shifting 0.5 px to align with px grid"
635                )
636                y = 0.5
637            if size[2] % 2 != 0:
638                print(
639                    f" Centered rounded cube z dimension is odd. Shifting 0.5 px to align with px grid"
640                )
641                z = 0.5
642
643        if size[0] == 0:
644            size = (0.0001, size[1], size[2])
645        if size[1] == 0:
646            size = (size[0], 0.0001, size[2])
647        if size[2] == 0:
648            size = (size[0], size[1], 0.0001)
649
650        radius = list(radius)
651        if radius[0] <= 0:
652            radius[0] = 0.0001
653        if radius[1] <= 0:
654            radius[1] = 0.0001
655        if radius[2] <= 0:
656            radius[2] = 0.0001
657
658        spheres = []
659        for i in range(2):
660            for j in range(2):
661                for k in range(2):
662                    if fn is None or fn < 0:
663                        s = Manifold.sphere(radius=1)
664                    else:
665                        s = Manifold.sphere(radius=1, circular_segments=fn)

667                        (
668                            radius[0],
669                            radius[1],
670                            radius[2],
671                        )
672                    )
673                    _x = (size[0] / 2 - radius[0]) if i else -(size[0] / 2 - radius[0])
674                    _y = (size[1] / 2 - radius[1]) if j else -(size[1] / 2 - radius[1])
675                    _z = (size[2] / 2 - radius[2]) if k else -(size[2] / 2 - radius[2])

679                            (y + _y),

685                            (
686                                size[0] / 2,
687                                size[1] / 2,
688                                size[2] / 2,
689                            )
690                        )
691                    spheres.append(s)
