136class ResinType:
137    def __init__(
138        self,
139        monomer: list[tuple[str, float]] = [("PEG", 100)],
140        uv_absorbers: list[tuple[str, float]] = [("NPS", 2.0)],
141        initiators: list[tuple[str, float]] = [("IRG", 1.0)],
142        additives: list[tuple[str, float]] = [],
143    ):
144        # Resin naming convention:
145        # Use 3 letter abbreviations for materials.
146        # Follow with a dash and the percent amount of the material
147        # If it's a monomer or oligimer, the percent is the fraction of total monomer/oligimer
148        # If it's an absorber, photoinitiator, or additive, the percent is a w/w fraction of the total resin mass
149        # When there are multiple materials in a category, separate them with a single underscore, _
150        # Separate categories of materials with two underscores, __
151        # Schema: MoA-XX_MoB-XX__AbA-XX_AbB-XX__PIA-XX_PIB_XX__AdA-XX_AdB-XX
152        # where:
153        #     MoA, MoB - monomers A and B
154        #     AbA, AbB - absorbers A and B
155        #     PIA, PIB - photoinitiators A and B
156        #     AdA, AdB - additives A and B
157        #     XX - number
158
159        if not isinstance(monomer, list) or not all(
160            (isinstance(x, tuple) or isinstance(x, list)) and len(x) == 2 for x in monomer
161        ):
162            raise ValueError("Monomer must be a list of tuples (name, percentage)")
163
164        if not isinstance(uv_absorbers, list) or not all(
165            (isinstance(x, tuple) or isinstance(x, list)) and len(x) == 2
166            for x in uv_absorbers
167        ):
168            raise ValueError("UV absorber must be a list of tuples (name, percentage)")
169
170        if not isinstance(initiators, list) or not all(
171            (isinstance(x, tuple) or isinstance(x, list)) and len(x) == 2
172            for x in initiators
173        ):
174            raise ValueError("Initiators must be a list of tuples (name, percentage)")
175
176        if not isinstance(additives, list) or not all(
177            (isinstance(x, tuple) or isinstance(x, list)) and len(x) == 2
178            for x in additives
179        ):
180            raise ValueError("Additives must be a list of tuples (name, percentage)")
181
182        if not all(
183            0 <= x[1] <= 100 for x in monomer + uv_absorbers + initiators + additives
184        ):
185            raise ValueError("All percentages must be between 0 and 100")
186        if sum(x[1] for x in monomer) != 100.0:
187            raise ValueError("Monomer percentages must add up to 100%")
188        if sum(x[1] for x in uv_absorbers + initiators + additives) > 100:
189            raise ValueError(
190                "UV absorber, initiators, and additives percentages must not exceed 100%"
191            )
192
193        self.monomer = monomer
194        self.uv_absorbers = uv_absorbers
195        self.initiators = initiators
196        self.additives = additives
197
198    def __str__(self):
199        # String matching schema
200        monomer_str = "_".join(
201            f"{name}-{percentage:.2f}" for name, percentage in self.monomer
202        )
203        uv_absorber_str = "_".join(
204            f"{name}-{percentage:.2f}" for name, percentage in self.uv_absorbers
205        )
206        initiators_str = "_".join(
207            f"{name}-{percentage:.2f}" for name, percentage in self.initiators
208        )
209        if len(self.additives) == 0:
210            return f"{monomer_str}__{uv_absorber_str}__{initiators_str}"
211        else:
212            additives_str = "_".join(
213                f"{name}-{percentage:.2f}" for name, percentage in self.additives
214            )
215            return f"{monomer_str}__{uv_absorber_str}__{initiators_str}__{additives_str}"

218class LightEngine:
219    def __init__(
220        self,
221        name: str = "visitech",
222        px_size: float = 0.0076,
223        px_count: tuple[int, int] = (2560, 1600),
224        wavelengths: list[int] = [365],
225        grayscale_available: list[bool] = [False],
226        origin: tuple[float, float] = (0.0, 0.0),  # Origin of image mm (x, y)
227    ):
228        if not isinstance(px_size, (int, float)) or px_size <= 0:
229            raise ValueError("Pixel size must be a positive number")
230        if (
231            not (isinstance(px_count, tuple) or isinstance(px_count, list))
232            or len(px_count) != 2
233            or not all(isinstance(x, int) or x <= 0 for x in px_count)
234        ):
235            raise ValueError("Pixel count must be a tuple of two positive integers")
236        if not isinstance(wavelengths, list) or not all(
237            isinstance(x, int) and x > 0 for x in wavelengths
238        ):
239            raise ValueError("Wavelengths must be a list of positive integers")
240        if not isinstance(grayscale_available, list) or not all(
241            isinstance(x, bool) for x in grayscale_available
242        ):
243            raise ValueError("Grayscale availability must be a list of booleans")
244        if (
245            not (isinstance(origin, tuple) or isinstance(origin, list))
246            or len(origin) != 2
247            or not all(isinstance(x, (int, float)) for x in origin)
248        ):
249            raise ValueError("Origin must be a tuple of two numbers (x, y)")
250        self.name = name
251        self.px_size = px_size
252        self.px_count = px_count
253        self.wavelengths = wavelengths
254        self.grayscale_available = grayscale_available
255        self.origin = origin
