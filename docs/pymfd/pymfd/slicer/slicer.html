 280    def _embed_image(self, pos, resolution, image_data, fqn):
 281        x = round(pos[0])
 282        y = round(pos[1])
 283
 284        slice_img = image_data
 285
 286        # Create a new empty image sized to the device
 287        slice_image = np.zeros((resolution[1], resolution[0]), dtype=np.uint8)
 288
 289        # Correct for numpy image origin (if you want origin at bottom-left)
 290        paste_y = resolution[1] - y
 291
 292        # compute paste coordinates (top-left y coordinate for the slice_img)
 293        top = paste_y - slice_img.shape[0]
 294        left = x
 295        bottom = top + slice_img.shape[0]
 296        right = left + slice_img.shape[1]
 297
 298        # Clip coordinates to image bounds to avoid exceptions
 299        top_clip = max(top, 0)
 300        left_clip = max(left, 0)
 301        bottom_clip = min(bottom, resolution[1])
 302        right_clip = min(right, resolution[0])
 303
 304        # compute corresponding region in slice_img
 305        src_top = top_clip - top if top < 0 else 0
 306        src_left = left_clip - left if left < 0 else 0
 307        src_bottom = src_top + (bottom_clip - top_clip)
 308        src_right = src_left + (right_clip - left_clip)
 309
 310        # Only paste if there's an overlap
 311        if bottom_clip > top_clip and right_clip > left_clip:
 312            try:
 313                slice_image[top_clip:bottom_clip, left_clip:right_clip] = slice_img[
 314                    src_top:src_bottom, src_left:src_right
 315                ]
 316            except Exception as e:
 317                print(
 318                    f"Warning: trouble pasting slice image for {fqn} at x={x},y={y}: {e}"
 319                )
 320
 321        else:
 322            print(
 323                f"Warning: slice image for {fqn} at x={x},y={y} is completely outside device bounds"
 324            )
 325            # still save an empty image or skip; here we'll skip
 326
 327        return slice_image

 566    def _match_or_find_closest_named_setting(
 567        self, settings, named_settings, ignore_keys=None
 568    ):
 569        if ignore_keys is None:
 570            ignore_keys = []
 571
 572        def dict_without_keys(d, keys):
 573            return {k: v for k, v in d.items() if k not in keys}
 574
 575        settings_filtered = dict_without_keys(settings, ignore_keys)
 576
 577        best_match_key = None
 578        fewest_differences = None
 579        differences_in_best = {}
 580
 581        for key, _settings in named_settings.items():
 582            _settings_filtered = dict_without_keys(_settings, ignore_keys)
 583
 584            if settings_filtered == _settings_filtered:
 585                # Exact match
 586                return key, {}
 587
 588            # Calculate differences
 589            differences = {
 590                k: settings_filtered.get(k)
 591                for k in set(settings_filtered) | set(_settings_filtered)
 592                if settings_filtered.get(k) != _settings_filtered.get(k)
 593            }
 594
 595            num_differences = len(differences)
 596
 597            if fewest_differences is None or num_differences < fewest_differences:
 598                best_match_key = key
 599                fewest_differences = num_differences
 600                differences_in_best = differences
 601
 602        return best_match_key, differences_in_best

 604    def _group_images_by_settings(self, slices):
 605        """
 606        Group images by their settings.
 607        This will return a list of slices where all settings match, except image file, exposure time, and the 2 waits.
 608        """

 615                continue
 616
 617            # Check if the current slice matches any of the existing groups
 618            match_found = False
 619            for group in grouped_slices:
 620                # Compare settings, ignoring image file, exposure time, and the 2 waits
 621                s1 = slice_info["exposure_settings"].to_dict()
 622                del s1["Image file"]
 623                del s1["Layer exposure time (ms)"]
 624                del s1["Wait before exposure (ms)"]
 625                del s1["Wait after exposure (ms)"]
 626
 627                s2 = group[0]["exposure_settings"].to_dict()
 628                del s2["Image file"]
 629                del s2["Layer exposure time (ms)"]
 630                del s2["Wait before exposure (ms)"]
 631                del s2["Wait after exposure (ms)"]
 632
 633                if s1 == s2:
 634                    group.append(slice_info)
 635                    match_found = True
 636                    break
 637
 638            if not match_found:

 680    def _compress_exposures(self, images, exposure_times, temp_directory):
 681        """
 682        Combine binary images into minimal exposure layers using the exposure-sum method.
 683        Optimized to avoid repeated min-searches and masking.
 684        """
 685
 686        def image_from_dict(slice_info):
 687            resolution = (
 688                int(slice_info["parent"].get_size()[0]),
 689                int(slice_info["parent"].get_size()[1]),
 690            )
 691            return self._embed_image(
 692                slice_info["position"],
 693                resolution,
 694                slice_info["image_data"],
 695                slice_info["device"].get_fully_qualified_name(),
 696            )
 697
 698        H = 0
 699        W = 0
 700        for image in images:
 701            if type(image) is not dict:
 702                H, W = image.shape
 703        if H == 0 or W == 0:
 704            H, W = image_from_dict(images[0]).shape
 705
 706        # mask = np.array(image_paths)
 707        # N, H, W = mask.shape
 708
 709        exposure_sum = np.zeros((H, W), dtype=float)
 710        for image, exp in zip(images, exposure_times):
 711            if type(image) is dict:
 712                img = image_from_dict(image)
 713            else:
 714                img = image
 715            exposure_sum[img == 255] += exp
 716
 717        # Find all unique nonzero exposures, sorted ascending
 718        unique_exposures = np.unique(exposure_sum[exposure_sum > 0])
 719        # print("Unique exposures:", unique_exposures)
 720        output_images = []
 721        output_exposures = []
 722
 723        prev = 0
 724        for exp in unique_exposures:
 725            # Mask for pixels with exposure >= exp
 726            layer_mask = exposure_sum >= exp
 727            out_img = layer_mask.astype(np.uint8) * 255
 728            output_images.append(out_img)
 729            output_exposures.append(exp - prev)
 730            prev = exp
 731
 732        return output_images, output_exposures

 281    def _embed_image(self, pos, resolution, image_data, fqn):
 282        x = round(pos[0])
 283        y = round(pos[1])
 284
 285        slice_img = image_data
 286
 287        # Create a new empty image sized to the device
 288        slice_image = np.zeros((resolution[1], resolution[0]), dtype=np.uint8)
 289
 290        # Correct for numpy image origin (if you want origin at bottom-left)
 291        paste_y = resolution[1] - y
 292
 293        # compute paste coordinates (top-left y coordinate for the slice_img)
 294        top = paste_y - slice_img.shape[0]
 295        left = x
 296        bottom = top + slice_img.shape[0]
 297        right = left + slice_img.shape[1]
 298
 299        # Clip coordinates to image bounds to avoid exceptions
 300        top_clip = max(top, 0)
 301        left_clip = max(left, 0)
 302        bottom_clip = min(bottom, resolution[1])
 303        right_clip = min(right, resolution[0])
 304
 305        # compute corresponding region in slice_img
 306        src_top = top_clip - top if top < 0 else 0
 307        src_left = left_clip - left if left < 0 else 0
 308        src_bottom = src_top + (bottom_clip - top_clip)
 309        src_right = src_left + (right_clip - left_clip)
 310
 311        # Only paste if there's an overlap
 312        if bottom_clip > top_clip and right_clip > left_clip:
 313            try:
 314                slice_image[top_clip:bottom_clip, left_clip:right_clip] = slice_img[
 315                    src_top:src_bottom, src_left:src_right
 316                ]
 317            except Exception as e:
 318                print(
 319                    f"Warning: trouble pasting slice image for {fqn} at x={x},y={y}: {e}"
 320                )
 321
 322        else:
 323            print(
 324                f"Warning: slice image for {fqn} at x={x},y={y} is completely outside device bounds"
 325            )
 326            # still save an empty image or skip; here we'll skip
 327
 328        return slice_image

 567    def _match_or_find_closest_named_setting(
 568        self, settings, named_settings, ignore_keys=None
 569    ):
 570        if ignore_keys is None:
 571            ignore_keys = []
 572
 573        def dict_without_keys(d, keys):
 574            return {k: v for k, v in d.items() if k not in keys}
 575
 576        settings_filtered = dict_without_keys(settings, ignore_keys)
 577
 578        best_match_key = None
 579        fewest_differences = None
 580        differences_in_best = {}
 581
 582        for key, _settings in named_settings.items():
 583            _settings_filtered = dict_without_keys(_settings, ignore_keys)
 584
 585            if settings_filtered == _settings_filtered:
 586                # Exact match
 587                return key, {}
 588
 589            # Calculate differences
 590            differences = {
 591                k: settings_filtered.get(k)
 592                for k in set(settings_filtered) | set(_settings_filtered)
 593                if settings_filtered.get(k) != _settings_filtered.get(k)
 594            }
 595
 596            num_differences = len(differences)
 597
 598            if fewest_differences is None or num_differences < fewest_differences:
 599                best_match_key = key
 600                fewest_differences = num_differences
 601                differences_in_best = differences
 602
 603        return best_match_key, differences_in_best

 681    def _compress_exposures(self, images, exposure_times, temp_directory):
 682        """
 683        Combine binary images into minimal exposure layers using the exposure-sum method.
 684        Optimized to avoid repeated min-searches and masking.
 685        """
 686
 687        def image_from_dict(slice_info):
 688            resolution = (
 689                int(slice_info["parent"].get_size()[0]),
 690                int(slice_info["parent"].get_size()[1]),
 691            )
 692            return self._embed_image(
 693                slice_info["position"],
 694                resolution,
 695                slice_info["image_data"],
 696                slice_info["device"].get_fully_qualified_name(),
 697            )
 698
 699        H = 0
 700        W = 0
 701        for image in images:
 702            if type(image) is not dict:
 703                H, W = image.shape
 704        if H == 0 or W == 0:
 705            H, W = image_from_dict(images[0]).shape
 706
 707        # mask = np.array(image_paths)
 708        # N, H, W = mask.shape
 709
 710        exposure_sum = np.zeros((H, W), dtype=float)
 711        for image, exp in zip(images, exposure_times):
 712            if type(image) is dict:
 713                img = image_from_dict(image)
 714            else:
 715                img = image
 716            exposure_sum[img == 255] += exp
 717
 718        # Find all unique nonzero exposures, sorted ascending
 719        unique_exposures = np.unique(exposure_sum[exposure_sum > 0])
 720        # print("Unique exposures:", unique_exposures)
 721        output_images = []
 722        output_exposures = []
 723
 724        prev = 0
 725        for exp in unique_exposures:
 726            # Mask for pixels with exposure >= exp
 727            layer_mask = exposure_sum >= exp
 728            out_img = layer_mask.astype(np.uint8) * 255
 729            output_images.append(out_img)
 730            output_exposures.append(exp - prev)
 731            prev = exp
 732
 733        return output_images, output_exposures
